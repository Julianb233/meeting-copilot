       ```python
           DELEGATE = "delegate"
           CHECK_DOMAIN = "check_domain"

       class MeetingTask(BaseModel):
           id: str
           title: str
           status: TaskStatus = TaskStatus.PENDING
           agent: str | None = None
           created_at: datetime = Field(default_factory=datetime.utcnow)
           completed_at: datetime | None = None
           result: str | None = None
           error: str | None = None


       class AgentStatus(BaseModel):
           name: str
           status: str = "idle"
       ```

    3. Create `engine/core/connection_manager.py` -- singleton shared between both servers:
       ```python
       """WebSocket connection manager -- shared singleton between ws_server and api_server.

       This module is the bridge that allows the REST API (port 8901) to see
       WebSocket connection state (port 8900). Both servers import `manager`
       from this module, and because they run in the same asyncio process,
       they share the same instance.
       """

       from __future__ import annotations
       import json
       import logging
       from typing import Any

       from fastapi import WebSocket

       from models import MeetingState, EngineConnectionAck, EnginePong

       logger = logging.getLogger("copilot.ws")


       class ConnectionManager:
           """Manages WebSocket connections and broadcasts."""

           def __init__(self) -> None:
               self.active_connections: list[WebSocket] = []
               self.state = MeetingState()

           async def connect(self, websocket: WebSocket) -> None:
               await websocket.accept()
               self.active_connections.append(websocket)
               logger.info(f"Client connected. Total: {len(self.active_connections)}")

               # Send full state on connect
               ack = EngineConnectionAck(meeting_state=self.state)
               await websocket.send_json(ack.model_dump(mode="json"))

           def disconnect(self, websocket: WebSocket) -> None:
               if websocket in self.active_connections:
                   self.active_connections.remove(websocket)
               logger.info(f"Client disconnected. Total: {len(self.active_connections)}")

           async def broadcast(self, message: dict[str, Any]) -> None:
               """Broadcast message to all connected clients."""
               disconnected: list[WebSocket] = []
               for connection in self.active_connections:
                   try:
                       await connection.send_json(message)
                   except Exception:
                       disconnected.append(connection)

               for conn in disconnected:
                   if conn in self.active_connections:
                       self.active_connections.remove(conn)

           async def handle_message(self, websocket: WebSocket, data: str) -> None:
               """Handle incoming message from panel."""
               try:
                   message = json.loads(data)
                   msg_type = message.get("type", "")

                   if msg_type == "ping":
                       pong = EnginePong()
                       await websocket.send_json(pong.model_dump())

                   elif msg_type == "quick_action":
                       action = message.get("action", "")
                       logger.info(f"Quick action received: {action}")
                       # TODO: Route to task orchestrator in Phase 3

                   else:
                       logger.warning(f"Unknown message type: {msg_type}")

               except json.JSONDecodeError:
                   logger.error(f"Invalid JSON received: {data[:100]}")


       # Module-level singleton -- both ws_server.py and api_server.py import this
       manager = ConnectionManager()
       ```

       ```python
       from fastapi import FastAPI, WebSocket, WebSocketDisconnect
       from fastapi.middleware.cors import CORSMiddleware

       import config
       from core.connection_manager import manager

       ws_app = FastAPI(title="Meeting Copilot WebSocket", version="0.1.0")

       ws_app.add_middleware(
           CORSMiddleware,
           allow_origins=[config.PANEL_ORIGIN],
           allow_credentials=True,
           allow_methods=["*"],
           allow_headers=["*"],
       )


       @ws_app.websocket("/ws")
       async def websocket_endpoint(websocket: WebSocket) -> None:
           await manager.connect(websocket)
           try:
               while True:
                   data = await websocket.receive_text()
                   await manager.handle_message(websocket, data)
           except WebSocketDisconnect:
               manager.disconnect(websocket)
       ```

    6. Create `engine/main.py` -- dual-server entrypoint using asyncio.gather:
       ```python
       """Meeting Copilot Engine -- runs WebSocket (8900) + REST API (8901) in one process.

       Architecture: Two separate FastAPI apps share state via a module-level
       ConnectionManager singleton. Both run as uvicorn servers in the same
       asyncio event loop using asyncio.gather.

       Pattern source: https://gist.github.com/tenuki/ff67f87cba5c4c04fd08d9c800437477
       """

       from __future__ import annotations
       import asyncio
       import logging
       import sys

       import uvicorn

       import config

       # Logging
       logging.basicConfig(
           level=logging.DEBUG if config.DEBUG else logging.INFO,
           format="%(asctime)s [%(name)s] %(levelname)s: %(message)s",
           stream=sys.stdout,
       )
       logger = logging.getLogger("copilot")


       async def run_server(app_path: str, port: int, name: str) -> None:
           """Run a uvicorn server for a FastAPI app."""
           server_config = uvicorn.Config(
               app_path,
               host=config.HOST,
               port=port,
               log_level="debug" if config.DEBUG else "info",
           )
           server = uvicorn.Server(server_config)
           logger.info(f"Starting {name} on {config.HOST}:{port}")
           await server.serve()


       async def main() -> None:
           logger.info("Meeting Copilot Engine starting...")
           logger.info(f"  WebSocket server (COM-01): port {config.WS_PORT}")
           logger.info(f"  REST API server (COM-02):  port {config.API_PORT}")
           logger.info(f"  CORS origin: {config.PANEL_ORIGIN}")
           logger.info(f"  Debug mode: {config.DEBUG}")

           await asyncio.gather(
               run_server("ws_server:ws_app", config.WS_PORT, "WebSocket"),
               run_server("api_server:api_app", config.API_PORT, "REST API"),
           )


       if __name__ == "__main__":
           asyncio.run(main())
       ```

    IMPORTANT ARCHITECTURE NOTE: This uses TWO separate FastAPI apps on TWO separate ports, NOT a single app. This matches requirements COM-01 (WebSocket on 8900) and COM-02 (REST on 8901). The apps share state through the singleton `manager` in `core/connection_manager.py` because they run in the same Python process.

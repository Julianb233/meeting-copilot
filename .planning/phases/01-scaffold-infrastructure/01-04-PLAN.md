    1. Create `deploy/nginx/` directory.

    2. Create `deploy/nginx/meeting-copilot.conf` -- nginx reverse proxy for dual-port architecture:
       ```nginx
       # Meeting Copilot Engine -- nginx reverse proxy
       # WebSocket on port 8900 (COM-01), REST API on port 8901 (COM-02)
       #
       # Install: sudo cp deploy/nginx/meeting-copilot.conf /etc/nginx/sites-available/meeting-copilot
       # Enable: sudo ln -s /etc/nginx/sites-available/meeting-copilot /etc/nginx/sites-enabled/
       # Test: sudo nginx -t && sudo systemctl reload nginx

       map $http_upgrade $connection_upgrade {
           default upgrade;
           ''      close;
       }

       server {
           listen 443 ssl;
           server_name copilot-api.agency.dev;

           # TLS -- update paths after certbot setup
           ssl_certificate /etc/letsencrypt/live/copilot-api.agency.dev/fullchain.pem;
           ssl_certificate_key /etc/letsencrypt/live/copilot-api.agency.dev/privkey.pem;
           ssl_protocols TLSv1.2 TLSv1.3;
           ssl_ciphers HIGH:!aNULL:!MD5;

           # WebSocket endpoint -- proxies to port 8900 (COM-01)
           location /ws {
               proxy_pass http://127.0.0.1:8900;
               proxy_http_version 1.1;
               proxy_set_header Upgrade $http_upgrade;
               proxy_set_header Connection $connection_upgrade;
               proxy_set_header Host $host;
               proxy_set_header X-Real-IP $remote_addr;
               proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
               proxy_set_header X-Forwarded-Proto $scheme;
               # 24 hours -- do NOT use default 60s, WebSocket will drop
               proxy_read_timeout 86400;
               proxy_send_timeout 86400;
               proxy_buffering off;
           }

           # REST API endpoint -- proxies to port 8901 (COM-02)
           location /api {
               proxy_pass http://127.0.0.1:8901;
               proxy_http_version 1.1;
               proxy_set_header Host $host;
               proxy_set_header X-Real-IP $remote_addr;
               proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
               proxy_set_header X-Forwarded-Proto $scheme;
           }

           # Also proxy /health and /state directly (api_server has no /api prefix)
           location /health {
               proxy_pass http://127.0.0.1:8901;
               proxy_set_header Host $host;
               proxy_set_header X-Real-IP $remote_addr;
           }

           location /state {
               proxy_pass http://127.0.0.1:8901;
               proxy_set_header Host $host;
               proxy_set_header X-Real-IP $remote_addr;
           }
       }

       # HTTP -> HTTPS redirect
       server {
           listen 80;
           server_name copilot-api.agency.dev;
           return 301 https://$host$request_uri;
       }
       ```

       NOTE: `copilot-api.agency.dev` is a placeholder domain. The actual domain needs a DNS A record pointing to the VPS IP and a certbot TLS cert.

    3. Create `deploy/meeting-copilot-engine.service` -- systemd unit file:
       ```ini
       [Unit]
       Description=Meeting Copilot Engine (WS:8900 + REST:8901)
       After=network.target

       [Service]
       Type=simple
       User=root
       WorkingDirectory=/opt/agency-workspace/meeting-copilot/engine
       ExecStart=/opt/agency-workspace/meeting-copilot/engine/.venv/bin/python main.py
       Restart=always
       RestartSec=5
       Environment=PYTHONUNBUFFERED=1
       EnvironmentFile=/opt/agency-workspace/meeting-copilot/engine/.env

       # Logging
       StandardOutput=journal
       StandardError=journal
       SyslogIdentifier=copilot-engine

       [Install]
       WantedBy=multi-user.target
       ```

    4. Create `deploy/deploy-engine.sh` -- deployment script:
       ```bash
       #!/usr/bin/env bash
       set -euo pipefail

       # Meeting Copilot Engine -- deploy to VPS
       # Usage: bash deploy/deploy-engine.sh

       ENGINE_DIR="/opt/agency-workspace/meeting-copilot/engine"
       SERVICE_NAME="meeting-copilot-engine"

       echo "=== Deploying Meeting Copilot Engine ==="

       # 1. Install/update Python deps
       echo "[1/4] Installing Python dependencies..."
       cd "$ENGINE_DIR"
       source .venv/bin/activate
       pip install -q -r requirements.txt

       # 2. Install systemd service
       echo "[2/4] Installing systemd service..."
       sudo cp /opt/agency-workspace/meeting-copilot/deploy/meeting-copilot-engine.service \
           /etc/systemd/system/$SERVICE_NAME.service
       sudo systemctl daemon-reload

       # 3. Install nginx config
       echo "[3/4] Installing nginx config..."
       sudo cp /opt/agency-workspace/meeting-copilot/deploy/nginx/meeting-copilot.conf \
           /etc/nginx/sites-available/meeting-copilot
       sudo ln -sf /etc/nginx/sites-available/meeting-copilot /etc/nginx/sites-enabled/meeting-copilot
       sudo nginx -t && sudo systemctl reload nginx

       # 4. Restart engine
       echo "[4/4] Restarting engine..."
       sudo systemctl enable $SERVICE_NAME
       sudo systemctl restart $SERVICE_NAME

       echo "=== Deployment complete ==="
       echo "Engine status: $(systemctl is-active $SERVICE_NAME)"
       echo "Health check: curl https://copilot-api.agency.dev/health"
       ```

    5. Make deploy script executable: `chmod +x deploy/deploy-engine.sh`

    def _extract_table_value(content: str, attribute: str) -> str | None:
        """Extract a value from a markdown table row like '| Formality | professional |'"""
        import re
        pattern = rf'\|\s*{re.escape(attribute)}\s*\|\s*(.+?)\s*\|'
        match = re.search(pattern, content, re.IGNORECASE)
        return match.group(1).strip() if match else None

    - meeting_type: from context.meeting_type (default "unknown")
    - summary_text: "Meeting: {title}\nType: {type}\nAttendees: {names}\nProjects: {projects}\n\nAction Items:\n- ...\n\nDecisions:\n- ...\n\nNext Steps:\n- ..."
